```typescript
import { LLMProvider, ProviderConfig, ProviderConfigError } from './types';

export async function createProvider(config: ProviderConfig): Promise<LLMProvider> {
  switch (config.provider) {
    case 'openrouter': {
      const apiKey = process.env.OPENROUTER_API_KEY;
      if (!apiKey) throw new ProviderConfigError('OPENROUTER_API_KEY is required', 'openrouter');
      // Dynamically import to avoid loading SDK until needed
      const { OpenRouterAdapter } = await import('./adapters/openrouter');
      return new OpenRouterAdapter(apiKey);
    }
    case 'azure': {
      const apiKey = process.env.AZURE_OPENAI_API_KEY;
      if (!apiKey) throw new ProviderConfigError('AZURE_OPENAI_API_KEY is required', 'azure');
      if (!config.resourceName) throw new ProviderConfigError('resourceName is required for azure provider', 'azure');
      if (!config.deploymentName) throw new ProviderConfigError('deploymentName is required for azure provider', 'azure');
      const { AzureOpenAIAdapter } = await import('./adapters/azure');
      return new AzureOpenAIAdapter(config.resourceName, config.deploymentName, apiKey, config.apiVersion);
    }
    case 'openai': {
      const apiKey = process.env.OPENAI_API_KEY;
      if (!apiKey) throw new ProviderConfigError('OPENAI_API_KEY is required', 'openai');
      const { OpenAIAdapter } = await import('./adapters/openai');
      return new OpenAIAdapter(apiKey);
    }
    case 'claude': {
      const apiKey = process.env.ANTHROPIC_API_KEY;
      if (!apiKey) throw new ProviderConfigError('ANTHROPIC_API_KEY is required', 'claude');
      const { ClaudeAdapter } = await import('./adapters/claude');
      return new ClaudeAdapter(apiKey);
    }
    default:
      throw new ProviderConfigError(`Unknown provider: ${(config as ProviderConfig).provider}`, (config as ProviderConfig).provider);
  }
}
```

Note: Dynamic imports are used because adapters are implemented in Plan 2 (Wave 2). Factory compiles and routes correctly even before adapters exist — TypeScript will complain about missing modules only at runtime, not compile time with dynamic imports.
