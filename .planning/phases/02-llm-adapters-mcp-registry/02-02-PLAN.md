```typescript
import { PricingRecord, PRICING_TABLE } from './table';

export function resolvePricing(provider: string, model: string): PricingRecord | null {
  return PRICING_TABLE[`${provider}:${model}`] ?? null;
}

export function estimateCostUsd(
  inputTokens: number,
  outputTokens: number,
  pricing: PricingRecord
): number {
  const cost = (inputTokens / 1_000_000) * pricing.inputPer1MTokens +
               (outputTokens / 1_000_000) * pricing.outputPer1MTokens;
  return Math.round(cost * 1_000_000) / 1_000_000; // 6 decimal places
}

export async function fetchOpenRouterPricing(): Promise<Map<string, PricingRecord>> {
  const apiKey = process.env.OPENROUTER_API_KEY;
  if (!apiKey) return new Map();

  try {
    const controller = new AbortController();
    const timeout = setTimeout(() => controller.abort(), 20_000);
    const res = await fetch('https://openrouter.ai/api/v1/models', {
      headers: { Authorization: `Bearer ${apiKey}` },
      signal: controller.signal,
    });
    clearTimeout(timeout);

    if (!res.ok) return new Map();
    const data = await res.json() as { data?: Array<{ id: string; pricing?: { prompt?: string; completion?: string } }> };
    const map = new Map<string, PricingRecord>();
    for (const model of data.data ?? []) {
      if (model.pricing?.prompt && model.pricing?.completion) {
        map.set(model.id, {
          inputPer1MTokens: parseFloat(model.pricing.prompt) * 1_000_000,
          outputPer1MTokens: parseFloat(model.pricing.completion) * 1_000_000,
        });
      }
    }
    return map;
  } catch {
    return new Map();
  }
}
```

```typescript
import OpenAI from 'openai';
import { LLMProvider, LLMRequest, LLMResponse, LLMChunk } from '../types';
import { resolvePricing, estimateCostUsd, fetchOpenRouterPricing } from '../../pricing/resolver';
import type { PricingRecord } from '../../pricing/table';

export class OpenRouterAdapter implements LLMProvider {
  private client: OpenAI;
  private pricingCache: Map<string, PricingRecord> = new Map();

  constructor(apiKey: string) {
    this.client = new OpenAI({
      apiKey,
      baseURL: 'https://openrouter.ai/api/v1',
      defaultHeaders: { 'HTTP-Referer': 'mcp-playground' },
    });
    // Fetch pricing in background (non-blocking)
    fetchOpenRouterPricing().then(cache => { this.pricingCache = cache; }).catch(() => {});
  }

  async complete(request: LLMRequest): Promise<LLMResponse> {
    const res = await this.client.chat.completions.create({
      model: request.model,
      messages: request.messages,
      max_tokens: request.max_tokens,
      temperature: request.temperature,
      top_p: request.top_p,
    });
    return {
      choices: res.choices.map(c => ({
        message: { role: c.message.role as 'user' | 'assistant' | 'system', content: c.message.content ?? '' },
        finish_reason: c.finish_reason ?? 'stop',
      })),
      usage: {
        prompt_tokens: res.usage?.prompt_tokens ?? 0,
        completion_tokens: res.usage?.completion_tokens ?? 0,
        total_tokens: res.usage?.total_tokens ?? 0,
      },
    };
  }

  async *stream(request: LLMRequest): AsyncIterable<LLMChunk> {
    const stream = await this.client.chat.completions.create({
      model: request.model,
      messages: request.messages,
      max_tokens: request.max_tokens,
      temperature: request.temperature,
      stream: true,
    });
    for await (const chunk of stream) {
      yield {
        delta: { content: chunk.choices[0]?.delta?.content ?? '' },
      };
    }
  }

  async estimateCost(inputTokens: number, outputTokens: number, model: string): Promise<number> {
    const cached = this.pricingCache.get(model);
    if (cached) return estimateCostUsd(inputTokens, outputTokens, cached);
    const static_ = resolvePricing('openrouter', model);
    if (static_) return estimateCostUsd(inputTokens, outputTokens, static_);
    const fallback = { inputPer1MTokens: 5, outputPer1MTokens: 15 };
    return estimateCostUsd(inputTokens, outputTokens, fallback);
  }
}
```

```typescript
import Anthropic from '@anthropic-ai/sdk';
import { LLMProvider, LLMRequest, LLMResponse, LLMChunk, LLMMessage } from '../types';
import { resolvePricing, estimateCostUsd } from '../../pricing/resolver';

export class ClaudeAdapter implements LLMProvider {
  private client: Anthropic;

  constructor(apiKey: string) {
    this.client = new Anthropic({ apiKey });
  }

  private extractMessages(messages: LLMMessage[]): { system: string; userMessages: Anthropic.MessageParam[] } {
    const systemMessages = messages.filter(m => m.role === 'system');
    const system = systemMessages.map(m => m.content).join('\n');
    const userMessages = messages
      .filter(m => m.role !== 'system')
      .map(m => ({ role: m.role as 'user' | 'assistant', content: m.content }));
    return { system, userMessages };
  }

  async complete(request: LLMRequest): Promise<LLMResponse> {
    const { system, userMessages } = this.extractMessages(request.messages);
    const res = await this.client.messages.create({
      model: request.model,
      system: system || undefined,
      messages: userMessages,
      max_tokens: request.max_tokens ?? 4096,
    });
    const content = res.content
      .filter(c => c.type === 'text')
      .map(c => (c as Anthropic.TextBlock).text)
      .join('');
    return {
      choices: [{ message: { role: 'assistant', content }, finish_reason: res.stop_reason ?? 'stop' }],
      usage: {
        prompt_tokens: res.usage.input_tokens,
        completion_tokens: res.usage.output_tokens,
        total_tokens: res.usage.input_tokens + res.usage.output_tokens,
      },
    };
  }

  async *stream(request: LLMRequest): AsyncIterable<LLMChunk> {
    const { system, userMessages } = this.extractMessages(request.messages);
    const stream = this.client.messages.stream({
      model: request.model,
      system: system || undefined,
      messages: userMessages,
      max_tokens: request.max_tokens ?? 4096,
    });
    for await (const event of stream) {
      if (event.type === 'content_block_delta' && event.delta.type === 'text_delta') {
        yield { delta: { content: event.delta.text } };
      }
    }
  }

  async estimateCost(inputTokens: number, outputTokens: number, model: string): Promise<number> {
    const pricing = resolvePricing('claude', model);
    if (pricing) return estimateCostUsd(inputTokens, outputTokens, pricing);
    return estimateCostUsd(inputTokens, outputTokens, { inputPer1MTokens: 3, outputPer1MTokens: 15 });
  }
}
```
