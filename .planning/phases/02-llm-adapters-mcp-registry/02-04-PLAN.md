Create `src/server/parser/translator.ts`:

```typescript
import { ASSERTION_PATTERNS } from './assertionPatterns';

export interface TranslatedAssertion {
  patternId: string | null;
  playwrightCall: string | null;
  original: string;
}

export function translateAssertion(thenClauseText: string): TranslatedAssertion {
  for (const pattern of ASSERTION_PATTERNS) {
    const match = pattern.regex.exec(thenClauseText);
    if (match) {
      return {
        patternId: pattern.id,
        playwrightCall: pattern.translate(match),
        original: thenClauseText,
      };
    }
  }
  return {
    patternId: null,
    playwrightCall: null,
    original: thenClauseText,
  };
}
```

```typescript
describe('translateAssertion', () => {
  it('translates URL pattern', () => {
    const result = translateAssertion('I should see the URL "https://example.com"');

    expect(result.playwrightCall).toBe('expect(page).toHaveURL("https://example.com")');
  });

    expect(result.patternId).toBe('title-match');
    expect(result.playwrightCall).toContain('toHaveTitle("Home")');
  });

  it('translates visibility pattern', () => {
    const result = translateAssertion('"Submit button" should be visible');
    expect(result.patternId).toBe('element-visible');
    expect(result.playwrightCall).toContain('toBeVisible()');
  });

  it('translates element text match', () => {
    const result = translateAssertion('the element "Welcome" should contain "user"');
    expect(result.patternId).toBe('element-text');
    expect(result.playwrightCall).toContain('toContainText("user")');
  });

  it('translates element count', () => {
    const result = translateAssertion('there should be 3 "item" elements');
    expect(result.patternId).toBe('element-count');
    expect(result.playwrightCall).toContain('toHaveCount(3)');
  });

  it('translates attribute check', () => {
    const result = translateAssertion('the "input" element should have attribute "type" with value "text"');
    expect(result.patternId).toBe('element-attribute');

  });

  it('translates form value check', () => {
    const result = translateAssertion('the input "email" should have value "test@example.com"');
    expect(result.patternId).toBe('form-value');
    expect(result.playwrightCall).toContain('toHaveValue');

  it('translates redirect pattern', () => {
```
